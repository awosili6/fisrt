# src/detectors/base_detector.py
from abc import ABC, abstractmethod
import torch
import numpy as np

class BaseDetector(ABC):
    """投毒检测器基类"""

    def __init__(self, model, tokenizer, device='cuda'):
        self.model = model
        self.tokenizer = tokenizer
        self.device = device

    @abstractmethod
    def detect(self, text: str, demonstrations: List[Dict] = None) -> Dict[str, Any]:
        """
        检测单个样本是否为投毒样本
        Returns:
            {
                'is_poisoned': bool,
                'score': float,      # 异常分数
                'confidence': float  # 置信度
            }
        """
        pass

    def batch_detect(self, texts: List[str], demonstrations: List[Dict] = None) -> List[Dict]:
        """批量检测"""
        return [self.detect(text, demonstrations) for text in texts]

# src/detectors/prompt_eraser.py
class PromptEraserDetector(BaseDetector):
    """基于提示擦除的检测器"""

    def __init__(self, model, tokenizer, erase_ratio=0.3, n_iterations=10,
                 aggregation='mean', device='cuda'):
        super().__init__(model, tokenizer, device)
        self.erase_ratio = erase_ratio
        self.n_iterations = n_iterations
        self.aggregation = aggregation

    def erase_tokens(self, text: str, erase_positions: List[int]) -> str:
        """擦除指定位置的token"""
        tokens = self.tokenizer.tokenize(text)
        erased_tokens = [t for i, t in enumerate(tokens) if i not in erase_positions]
        return self.tokenizer.convert_tokens_to_string(erased_tokens)

    def compute_prediction(self, text: str, demonstrations: List[Dict] = None) -> torch.Tensor:
        """获取模型的预测分布"""
        # 构建完整的提示（演示示例 + 查询样本）
        if demonstrations:
            prompt = self._build_prompt(demonstrations, text)
        else:
            prompt = text

        inputs = self.tokenizer(prompt, return_tensors='pt').to(self.device)
        with torch.no_grad():
            outputs = self.model(**inputs)
            logits = outputs.logits[0, -1, :]  # 最后一个token的logits
            probs = torch.softmax(logits, dim=-1)
        return probs

    def compute_similarity(self, pred1: torch.Tensor, pred2: torch.Tensor) -> float:
        """计算两个预测分布的相似度"""
        # 可以使用KL散度、JS散度或余弦相似度
        kl_div = torch.sum(pred1 * torch.log(pred1 / (pred2 + 1e-10) + 1e-10))
        return -kl_div.item()  # 转换为相似度（越大越相似）

    def detect(self, text: str, demonstrations: List[Dict] = None) -> Dict[str, Any]:
        """
        核心检测逻辑：通过多次擦除计算稳定性分数
        """
        # 1. 获取原始预测
        original_pred = self.compute_prediction(text, demonstrations)

        # 2. 获取token数量
        tokens = self.tokenizer.tokenize(text)
        n_tokens = len(tokens)
        n_erase = max(1, int(n_tokens * self.erase_ratio))

        # 3. 多次随机擦除并计算预测变化
        stability_scores = []
        for _ in range(self.n_iterations):
            # 随机选择擦除位置
            erase_positions = random.sample(range(n_tokens), n_erase)
            erased_text = self.erase_tokens(text, erase_positions)

            # 计算擦除后的预测
            erased_pred = self.compute_prediction(erased_text, demonstrations)

            # 计算相似度
            sim = self.compute_similarity(original_pred, erased_pred)
            stability_scores.append(sim)

        # 4. 聚合分数
        if self.aggregation == 'mean':
            final_score = np.mean(stability_scores)
        elif self.aggregation == 'median':
            final_score = np.median(stability_scores)
        elif self.aggregation == 'min':
            final_score = np.min(stability_scores)
        else:
            final_score = np.mean(stability_scores)

        # 5. 判断是否投毒（分数越低，越可能是投毒样本）
        # 需要预先确定阈值，或通过验证集学习阈值
        threshold = self._get_threshold()
        is_poisoned = final_score < threshold

        return {
            'is_poisoned': is_poisoned,
            'score': final_score,
            'stability_scores': stability_scores,
            'confidence': abs(final_score - threshold) / threshold
        }

    def _get_threshold(self) -> float:
        """获取检测阈值（可通过验证集学习）"""
        # 默认阈值，实际使用时应该在验证集上调参
        return -0.5  # KL散度为负值表示相似度

# src/detectors/attention_eraser.py
class AttentionEraserDetector(PromptEraserDetector):
    """基于Attention权重的擦除检测器"""

    def __init__(self, model, tokenizer, erase_ratio=0.3, n_iterations=10,
                 attention_layer=-1, device='cuda'):
        super().__init__(model, tokenizer, erase_ratio, n_iterations, device=device)
        self.attention_layer = attention_layer

    def get_attention_weights(self, text: str) -> torch.Tensor:
        """获取输入各token的attention权重"""
        inputs = self.tokenizer(text, return_tensors='pt').to(self.device)

        with torch.no_grad():
            outputs = self.model(**inputs, output_attentions=True)
            attentions = outputs.attentions[self.attention_layer]  # [batch, heads, seq, seq]

        # 计算每个token的平均attention权重
        avg_attention = attentions.mean(dim=(0, 1, 2))  # [seq_len]
        return avg_attention[1:-1]  # 去掉CLS和SEP

    def detect(self, text: str, demonstrations: List[Dict] = None) -> Dict[str, Any]:
        """基于attention权重选择擦除位置"""
        tokens = self.tokenizer.tokenize(text)
        n_tokens = len(tokens)
        n_erase = max(1, int(n_tokens * self.erase_ratio))

        # 获取attention权重
        attention_weights = self.get_attention_weights(text)

        # 根据权重选择擦除位置（可以选权重高的或权重低的）
        # 策略1: 擦除高权重token（关键信息）
        # 策略2: 擦除低权重token（噪音信息）

        # 这里实现策略2: 擦除低权重token
        _, erase_positions = torch.topk(attention_weights, n_erase, largest=False)
        erase_positions = erase_positions.cpu().tolist()

        # 调用父类的检测逻辑
        return super().detect_with_positions(text, demonstrations, erase_positions)

class STRIPDetector(BaseDetector):
    """
    STRIP检测器基线实现
    参考: Gao et al. "STRIP: A Defence Against Trojan Attacks on Deep Neural Networks"
    """

    def __init__(self, model, tokenizer, n_iterations=100, device='cuda'):
        super().__init__(model, tokenizer, device)
        self.n_iterations = n_iterations

    def perturb_input(self, text: str) -> str:
        """对输入进行扰动（同义词替换、插入、删除）"""
        # 实现文本扰动策略
        pass

    def compute_entropy(self, predictions: List[torch.Tensor]) -> float:
        """计算预测分布的熵"""
        avg_pred = torch.stack(predictions).mean(dim=0)
        entropy = -torch.sum(avg_pred * torch.log(avg_pred + 1e-10))
        return entropy.item()

    def detect(self, text: str, demonstrations: List[Dict] = None) -> Dict[str, Any]:
        """
        核心思想: 投毒样本对扰动更敏感，预测熵更低
        """
        predictions = []
        for _ in range(self.n_iterations):
            perturbed = self.perturb_input(text)
            pred = self.compute_prediction(perturbed, demonstrations)
            predictions.append(pred)

        entropy = self.compute_entropy(predictions)

        # 熵越低，越可能是投毒样本
        threshold = self._get_threshold()
        is_poisoned = entropy < threshold

        return {
            'is_poisoned': is_poisoned,
            'score': entropy,
            'confidence': abs(entropy - threshold) / threshold if threshold > 0 else 0.5
        }

# src/detectors/baselines/onion_detector.py
class ONIONDetector(BaseDetector):
    """
    ONION检测器基线实现
    参考: Qi et al. "ONION: A Simple and Effective Defense Against Textual Backdoor Attacks"
    """

    def __init__(self, model, tokenizer, perplexity_threshold=None, device='cuda'):
        super().__init__(model, tokenizer, device)
        self.perplexity_threshold = perplexity_threshold

    def compute_perplexity(self, text: str) -> float:
        """计算文本的困惑度"""
        inputs = self.tokenizer(text, return_tensors='pt').to(self.device)

        with torch.no_grad():
            outputs = self.model(**inputs, labels=inputs['input_ids'])
            loss = outputs.loss

        perplexity = torch.exp(loss).item()
        return perplexity

    def detect(self, text: str, demonstrations: List[Dict] = None) -> Dict[str, Any]:
        """
        核心思想: 如果某个词显著增加困惑度，可能是触发词
        """
        words = text.split()
        original_ppl = self.compute_perplexity(text)

        word_scores = []
        for i in range(len(words)):
            # 移除第i个词计算困惑度
            removed_text = " ".join(words[:i] + words[i+1:])
            removed_ppl = self.compute_perplexity(removed_text)
            score = removed_ppl - original_ppl  # 困惑度下降越大，该词越可疑
            word_scores.append((i, words[i], score))

        # 如果存在显著降低困惑度的词，可能是触发词
        min_score = min(s for _, _, s in word_scores)
        is_poisoned = min_score < -self.perplexity_threshold if self.perplexity_threshold else min_score < -1.0

        return {
            'is_poisoned': is_poisoned,
            'score': min_score,
            'suspicious_word': word_scores[0][1] if is_poisoned else None,
            'confidence': abs(min_score) / abs(self.perplexity_threshold) if self.perplexity_threshold else 0.5
        }

# src/models/llm_wrapper.py
import torch
from transformers import AutoModelForCausalLM, AutoTokenizer, AutoModelForSeq2SeqLM

class LLMWrapper:
    """大语言模型统一封装"""

    def __init__(self, model_name: str, device: str = 'cuda', load_in_8bit: bool = False):
        self.model_name = model_name
        self.device = device
        self.load_in_8bit = load_in_8bit

        self._load_model()

    def _load_model(self):
        """加载模型和tokenizer"""
        self.tokenizer = AutoTokenizer.from_pretrained(self.model_name, trust_remote_code=True)

        if self.tokenizer.pad_token is None:
            self.tokenizer.pad_token = self.tokenizer.eos_token

        # 根据模型类型选择加载方式
        if 'chatglm' in self.model_name.lower():
            self.model = AutoModelForSeq2SeqLM.from_pretrained(
                self.model_name,
                trust_remote_code=True,
                torch_dtype=torch.float16 if not self.load_in_8bit else torch.int8,
                device_map='auto' if self.load_in_8bit else None
            )
        else:
            self.model = AutoModelForCausalLM.from_pretrained(
                self.model_name,
                trust_remote_code=True,
                torch_dtype=torch.float16 if not self.load_in_8bit else torch.int8,
                device_map='auto' if self.load_in_8bit else None
            )

        if not self.load_in_8bit:
            self.model = self.model.to(self.device)

        self.model.eval()

    def predict(self, prompt: str, max_new_tokens: int = 10) -> str:
        """生成预测结果"""
        inputs = self.tokenizer(prompt, return_tensors='pt', padding=True).to(self.device)

        with torch.no_grad():
            outputs = self.model.generate(
                **inputs,
                max_new_tokens=max_new_tokens,
                do_sample=False,
                pad_token_id=self.tokenizer.pad_token_id
            )

        generated_text = self.tokenizer.decode(outputs[0], skip_special_tokens=True)
        # 移除原始prompt，只保留生成部分
        response = generated_text[len(self.tokenizer.decode(inputs['input_ids'][0], skip_special_tokens=True)):]
        return response.strip()

    def get_hidden_states(self, text: str, layer: int = -1) -> torch.Tensor:
        """获取隐藏层表示（用于RAP等基于表征的检测方法）"""
        inputs = self.tokenizer(text, return_tensors='pt').to(self.device)

        with torch.no_grad():
            outputs = self.model(**inputs, output_hidden_states=True)
            hidden_states = outputs.hidden_states[layer]  # [batch, seq, hidden]

        return hidden_states.mean(dim=1)  # [batch, hidden]

class DatasetLoader:

    @staticmethod
    def load(dataset_name: str, max_samples: int = None) -> Dict[str, List]:

        elif dataset_name == 'ag_news':

        elif dataset_name == 'hate_speech':
            return DatasetLoader._load_hate_speech(max_samples)
        elif dataset_name == 'trec':
            return DatasetLoader._load_trec(max_samples)
        else:
            raise ValueError(f"Unsupported dataset: {dataset_name}")

    @staticmethod
    def _load_sst2(max_samples: int = None) -> Dict[str, List]:

        train_texts = dataset['train']['sentence']

        test_labels = dataset['validation']['label']

            train_labels = train_labels[:max_samples]
            test_texts = test_texts[:max_samples//5]
            test_labels = test_labels[:max_samples//5]

            'train': {'texts': train_texts, 'labels': train_labels},
            'test': {'texts': test_texts, 'labels': test_labels}

class PoisonedDataset:
    """投毒数据集封装"""

    def __init__(self, texts: List[str], labels: List[int],
                 attack: 'BaseAttack' = None, demonstrations: List[Dict] = None):
        self.texts = texts
        self.labels = labels
        self.attack = attack
        self.demonstrations = demonstrations
        self.poison_indices = []

        if attack:
            self._apply_attack()

    def _apply_attack(self):
        """应用投毒攻击"""
        n_poison = int(len(self.texts) * self.attack.poison_rate)
        self.poison_indices = random.sample(range(len(self.texts)), n_poison)

        for idx in self.poison_indices:
            self.texts[idx] = self.attack.inject_trigger(self.texts[idx])
            self.labels[idx] = self.attack.target_label

    def create_icl_prompt(self, query_idx: int, n_shots: int = 5) -> str:
        """
        创建上下文学习提示
        格式: [示例1]\n[示例2]\n...\n[查询样本] ->
        """
        # 从训练集中采样n_shots个示例
        demo_indices = random.sample(
            [i for i in range(len(self.texts)) if i != query_idx],
            n_shots
        )

        demonstrations = []
        for idx in demo_indices:
            demonstrations.append({
                'text': self.texts[idx],
                'label': self.labels[idx],
                'is_poisoned': idx in self.poison_indices
            })

        # 构建prompt
        prompt_parts = []
        for demo in demonstrations:
            prompt_parts.append(f"Text: {demo['text']}\nLabel: {demo['label']}")

        prompt_parts.append(f"Text: {self.texts[query_idx]}\nLabel:")

        return '\n\n'.join(prompt_parts), demonstrations

# src/evaluation/metrics.py
from typing import List, Dict
from sklearn.metrics import accuracy_score, precision_recall_fscore_support, confusion_matrix
import numpy as np
import time

class Evaluator:
    """评估器"""

    @staticmethod
    def compute_classification_metrics(y_true: List[int], y_pred: List[int]) -> Dict:
        """计算分类指标"""
        accuracy = accuracy_score(y_true, y_pred)
        precision, recall, f1, _ = precision_recall_fscore_support(
            y_true, y_pred, average='binary', zero_division=0
        )

        cm = confusion_matrix(y_true, y_pred)
        tn, fp, fn, tp = cm.ravel()

        fpr = fp / (fp + tn) if (fp + tn) > 0 else 0  # 误检率
        fnr = fn / (fn + tp) if (fn + tp) > 0 else 0  # 漏检率

        return {
            'accuracy': accuracy,
            'precision': precision,
            'recall': recall,
            'f1_score': f1,
            'fpr': fpr,
            'fnr': fnr,
            'confusion_matrix': cm.tolist()
        }

    @staticmethod
    def compute_attack_metrics(clean_labels: List[int], clean_preds: List[int],
                               poison_labels: List[int], poison_preds: List[int]) -> Dict:
        """计算攻击效果指标"""
        cacc = accuracy_score(clean_labels, clean_preds)  # 干净准确率
        asr = accuracy_score(poison_labels, poison_preds)  # 攻击成功率

        return {
            'CACC': cacc,
            'ASR': asr,
            'fidelity': cacc / (asr + 1e-10)  # 保真度
        }

    @staticmethod
    def evaluate_detector(detector, test_texts: List[str],
                          true_labels: List[int],  # 1表示投毒，0表示干净
                          demonstrations: List[Dict] = None) -> Dict:
        """评估检测器性能"""
        predictions = []
        scores = []
        latencies = []

        for text in test_texts:
            start_time = time.time()
            result = detector.detect(text, demonstrations)
            latency = time.time() - start_time

            predictions.append(1 if result['is_poisoned'] else 0)
            scores.append(result['score'])
            latencies.append(latency)

        metrics = Evaluator.compute_classification_metrics(true_labels, predictions)
        metrics['avg_latency'] = np.mean(latencies) * 1000  # 转换为ms
        metrics['throughput'] = 1.0 / np.mean(latencies) if np.mean(latencies) > 0 else 0

        return metrics

    def plot_sensitivity_analysis(param_values: List, f1_scores: List,
                                  latencies: List, param_name: str = 'Parameter',
                                  save_path: str = None):
        """绘制参数敏感性分析图"""
        fig, ax1 = plt.subplots(figsize=(10, 6))

        color1 = 'tab:blue'
        ax1.set_xlabel(param_name)
        ax1.set_ylabel('F1-Score', color=color1)
        ax1.plot(param_values, f1_scores, color=color1, marker='o', label='F1-Score')
        ax1.tick_params(axis='y', labelcolor=color1)

        ax2 = ax1.twinx()
        color2 = 'tab:red'
        ax2.set_ylabel('Latency (ms)', color=color2)
        ax2.plot(param_values, latencies, color=color2, marker='s', label='Latency')
        ax2.tick_params(axis='y', labelcolor=color2)

        fig.tight_layout()

        if save_path:
            plt.savefig(save_path)
        plt.close()

    @staticmethod
    def plot_pareto_frontier(f1_scores: List[float], latencies: List[float],
                             labels: List[str] = None, save_path: str = None):
        """绘制Pareto前沿曲线（效率-准确性权衡）"""
        plt.figure(figsize=(10, 6))

        # 绘制所有点
        plt.scatter(latencies, f1_scores, s=100, alpha=0.6)

        # 标注每个点
        if labels:
            for i, label in enumerate(labels):
                plt.annotate(label, (latencies[i], f1_scores[i]),
                            xytext=(5, 5), textcoords='offset points', fontsize=8)

        plt.xlabel('Latency (ms)')
        plt.ylabel('F1-Score')
        plt.title('Efficiency-Accuracy Trade-off (Pareto Frontier)')
        plt.grid(True, alpha=0.3)

        if save_path:
            plt.savefig(save_path)
        plt.close()

"""
实验1: 攻击复现实验
验证不同攻击方法在不同模型和数据集上的效果
"""

import argparse
import json
import os
from tqdm import tqdm

from src.models.llm_wrapper import LLMWrapper
from src.datasets.data_loader import DatasetLoader, PoisonedDataset
from src.attacks.badnets_attack import BadNetsAttack
from src.attacks.insert_sent_attack import InsertSentAttack
from src.attacks.syntactic_attack import SyntacticAttack
from src.evaluation.metrics import Evaluator

def run_attack_experiment(model_name: str, dataset_name: str, attack_type: str,
                          poison_rate: float, output_dir: str):
    """运行单组攻击实验"""

    # 1. 加载模型
    print(f"Loading model: {model_name}")
    model = LLMWrapper(model_name, load_in_8bit=True)

    # 2. 加载数据集
    print(f"Loading dataset: {dataset_name}")
    data = DatasetLoader.load(dataset_name, max_samples=1000)

    # 3. 初始化攻击
    target_label = 0  # 假设目标标签为0
    if attack_type == 'badnets':
        attack = BadNetsAttack(trigger='cf', target_label=target_label, poison_rate=poison_rate)
    elif attack_type == 'insertsent':
        attack = InsertSentAttack(trigger='I watched this 3D movie', target_label=target_label, poison_rate=poison_rate)
    elif attack_type == 'syntactic':
        attack = SyntacticAttack(trigger='S(SBAR)(,)(NP)(VP)(.)', target_label=target_label, poison_rate=poison_rate)
    else:
        raise ValueError(f"Unknown attack type: {attack_type}")

    # 4. 构造投毒数据集
    print(f"Applying {attack_type} attack with poison rate {poison_rate}")
    poisoned_dataset = PoisonedDataset(
        data['train']['texts'],
        data['train']['labels'],
        attack=attack
    )

    # 5. 评估攻击效果
    print("Evaluating attack...")
    clean_preds = []
    poison_preds = []

    # 评估干净样本
    for text, label in tqdm(zip(data['test']['texts'][:100], data['test']['labels'][:100]),
                            desc="Clean samples"):
        # 构建ICL prompt
        prompt, _ = poisoned_dataset.create_icl_prompt(0, n_shots=5)
        pred_text = model.predict(prompt)
        # 解析预测结果
        pred_label = int(pred_text.strip()) if pred_text.strip().isdigit() else -1
        clean_preds.append(pred_label)

    # 评估投毒样本
    poisoned_test_texts = [attack.inject_trigger(t) for t in data['test']['texts'][:100]]
    for text in tqdm(poisoned_test_texts, desc="Poisoned samples"):
        prompt, _ = poisoned_dataset.create_icl_prompt(0, n_shots=5)
        pred_text = model.predict(prompt)
        pred_label = int(pred_text.strip()) if pred_text.strip().isdigit() else -1
        poison_preds.append(pred_label)

    # 6. 计算指标
    metrics = Evaluator.compute_attack_metrics(
        data['test']['labels'][:100], clean_preds,
        [target_label] * 100, poison_preds
    )

    print(f"Results: CACC={metrics['CACC']:.3f}, ASR={metrics['ASR']:.3f}")

    # 7. 保存结果
    os.makedirs(output_dir, exist_ok=True)
    result_file = os.path.join(output_dir, f"{model_name.replace('/', '_')}_{dataset_name}_{attack_type}_{poison_rate}.json")
    with open(result_file, 'w') as f:
        json.dump(metrics, f, indent=2)

    return metrics

if __name__ == '__main__':
    parser = argparse.ArgumentParser()
    parser.add_argument('--model', type=str, default='meta-llama/Llama-2-7b-chat-hf')
    parser.add_argument('--dataset', type=str, default='sst2')
    parser.add_argument('--attack', type=str, default='badnets', choices=['badnets', 'insertsent', 'syntactic'])
    parser.add_argument('--poison-rate', type=float, default=0.1)
    parser.add_argument('--output-dir', type=str, default='results/attack_reproduction')
    args = parser.parse_args()

    run_attack_experiment(args.model, args.dataset, args.attack, args.poison_rate, args.output_dir)

def run_detection_experiment(model_name: str, dataset_name: str, detector_type: str,

    # 2. 加载数据集并投毒

        detector = PromptEraserDetector(

            erase_ratio=erase_ratio, n_iterations=n_iterations
        )

        detector = AttentionEraserDetector(

            erase_ratio=erase_ratio, n_iterations=n_iterations
        )

        detector = STRIPDetector(model.model, model.tokenizer, n_iterations=n_iterations)

    else:
        raise ValueError(f"Unknown detector type: {detector_type}")

    # 4. 准备测试集（混合干净和投毒样本）
    test_texts = data['test']['texts'][:50]
    test_labels = [0] * 50  # 干净样本标签为0

    print("Evaluating detector...")
    metrics = Evaluator.evaluate_detector(detector, test_texts, test_labels)

    print(f"Results: F1={metrics['f1_score']:.3f}, Accuracy={metrics['accuracy']:.3f}, "
          f"Latency={metrics['avg_latency']:.1f}ms")

    # 6. 保存结果

def run_sensitivity_analysis(model_name: str, dataset_name: str,
                             param_name: str, param_values: list,
                             output_dir: str):

    # 1. 加载模型和数据
    model = LLMWrapper(model_name, load_in_8bit=True)
    data = DatasetLoader.load(dataset_name, max_samples=300)

                                            erase_ratio=value, n_iterations=10)

            detector = PromptEraserDetector(model.model, model.tokenizer,
                                            erase_ratio=0.3, n_iterations=value)

            raise ValueError(f"Unknown parameter: {param_name}")

        results.append({
            'param_value': value,
            'f1_score': metrics['f1_score'],
            'accuracy': metrics['accuracy'],
            'fpr': metrics['fpr'],
