## 或使用 Git 下载（推荐，最可靠）
# 先安装 git-lfs: https://git-lfs.github.com/
# 然后运行:
git lfs install
git clone https://hf-mirror.com/distilgpt2 ./models/distilgpt2--distilgpt2

## 或使用 aria2c 多线程下载工具
# 安装 aria2c 后运行：
aria2c -c -x 4 -s 4 -d ./models/distilgpt2--distilgpt2/ "https://hf-mirror.com/distilgpt2/resolve/main/model.safetensors"
aria2c -c -x 4 -s 4 -d ./models/distilgpt2--distilgpt2/ "https://hf-mirror.com/distilgpt2/resolve/main/config.json"
aria2c -c -x 4 -s 4 -d ./models/distilgpt2--distilgpt2/ "https://hf-mirror.com/distilgpt2/resolve/main/tokenizer_config.json"
aria2c -c -x 4 -s 4 -d ./models/distilgpt2--distilgpt2/ "https://hf-mirror.com/distilgpt2/resolve/main/vocab.json"
aria2c -c -x 4 -s 4 -d ./models/distilgpt2--distilgpt2/ "https://hf-mirror.com/distilgpt2/resolve/main/merges.txt"
aria2c -c -x 4 -s 4 -d ./models/distilgpt2--distilgpt2/ "https://hf-mirror.com/distilgpt2/resolve/main/tokenizer.json"
aria2c -c -x 4 -s 4 -d ./models/distilgpt2--distilgpt2/ "https://hf-mirror.com/distilgpt2/resolve/main/special_tokens_map.json"
aria2c -c -x 4 -s 4 -d ./models/distilgpt2--distilgpt2/ "https://hf-mirror.com/distilgpt2/resolve/main/generation_config.json"
